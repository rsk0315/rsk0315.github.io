Examples

use nekolib::traits::DisjointSet;
use nekolib::ds::UnionFind;

let mut uf = UnionFind::new(4);
assert!(!uf.equiv(0, 2));
uf.unite(0, 1);
uf.unite(1, 2);
assert!(uf.equiv(0, 2));
assert!(!uf.equiv(0, 3));
assert_eq!(uf.count(0), 3);
