fn div63(rdi: u64) -> u64 {
    let rdx = ((rdi as u128 * 0x410410410410411_u128) >> 64) as u64;
    (((rdi - rdx) >> 1) + rdx) >> 5
}

fn div64(rdi: u64) -> u64 { rdi >> 6 }

fn div65(rdi: u64) -> u64 {
    ((rdi as u128 * 0xFC0FC0FC0FC0FC1_u128) >> 66) as u64
}

for i in 0..=100000 {

}
